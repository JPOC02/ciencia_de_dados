clone o repo

git clone https://github.com/JPOC02/ciencia_de_dados


entre no repo

cd ciência_de_dados

python -m venv venv

source venv/Scripts/activate <- no Windows git bash
source venv/bin/activate <- no linux

pip install -r requirements.txt
